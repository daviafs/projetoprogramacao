float showA (){
  
  float matrixA[20][20];
  int i, j, iA, jA;
  
  iA=5;
  jA=5;
  
  srand(time(0));

  for(i=0; i<iA; i++){
    for(j=0; j<jA; j++){
      matrixA[i][j] = rand()%100;
    }
  }
  
  printf("MATRIX A:\n");
  for(i=0;i<iA;i++){
    for(j=0;j<jA;j++){
      printf("%2.2f ", matrixA[i][j]);
    }
    printf("\n");
  }
}
